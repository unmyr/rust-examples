== Examples

=== duplicate definitions for `get_message`

==== Single file

[source,rust]
.src/lib.rs
----
pub struct Hello {
    pub msg: String
}

pub mod v1 {
    impl super::Hello {
        pub fn get_message() -> String {
            String::from("v1: Hello world")
        }
    }
}

pub mod v2 {
    impl super::Hello {
        pub fn get_message() -> String {
            String::from("v2: Hello world")
        }
    }
}
----
