=== Using a match expression to handle the Result variants that might be returned
[source,rust]
.Using a match expression to handle the Result variants that might be returned
----
use std::fs::File;

fn main() {
    let f = File::open("hello.txt");

    let f = match f {
        Ok(file) => file,
        Err(error) => panic!("Problem opening the file: {:?}", error),
    };
}
----
